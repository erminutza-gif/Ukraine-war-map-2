<head>
  <meta charset="UTF-8" />
  <title>Ukraine War Map</title>
  <link
    rel="stylesheet"
    href="https://unpkg.com/leaflet@1.9.4/dist/leaflet.css"
  />
  <style>
    #map { height: 100vh; }
    .legend {
      background: white;
      padding: 8px;
      font-size: 12px;
      line-height: 1.4;
    }
    .legend span {
      display: inline-block;
      width: 12px;
      height: 12px;
      margin-right: 4px;
    }
  </style>
</head>
<body>
<div id="map"></div>

<script src="https://unpkg.com/leaflet@1.9.4/dist/leaflet.js"></script>
<script>
  // 1. Base map
  const map = L.map('map').setView([48.5, 31.0], 6); // roughly center of Ukraine

  L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {
    maxZoom: 10,
    attribution: '&copy; OpenStreetMap'
  }).addTo(map);

  // 2. Territorial control (dummy polygons)
  const territorialGeojson = {
    "type": "FeatureCollection",
    "features": [
      {
        "type": "Feature",
        "properties": { "status": "occupied", "name": "Example area" },
        "geometry": {
          "type": "Polygon",
          "coordinates": [[[30, 48],[31, 48],[31, 49],[30, 49],[30, 48]]]
        }
      },
      {
        "type": "Feature",
        "properties": { "status": "liberated", "name": "Example area 2" },
        "geometry": {
          "type": "Polygon",
          "coordinates": [[[32, 48],[33, 48],[33, 49],[32, 49],[32, 48]]]
        }
      }
    ]
  };

  function territorialStyle(feature) {
    const status = feature.properties.status;
    let color = '#999';
    if (status === 'occupied') color = '#d73027';
    if (status === 'liberated') color = '#1a9850';
    if (status === 'contested') color = '#fdae61';
    return {
      color: '#333',
      weight: 1,
      fillColor: color,
      fillOpacity: 0.6
    };
  }

  const territorialLayer = L.geoJSON(territorialGeojson, {
    style: territorialStyle,
    onEachFeature: (feature, layer) => {
      layer.bindPopup(
        `<b>${feature.properties.name}</b><br>Status: ${feature.properties.status}`
      );
    }
  }).addTo(map);

  // 3. Air alerts (dummy oblast polygons)
  const alertsGeojson = {
    "type": "FeatureCollection",
    "features": [
      {
        "type": "Feature",
        "properties": { "type": "missile", "region": "Dummy oblast" },
        "geometry": {
          "type": "Polygon",
          "coordinates": [[[29, 47.5],[34, 47.5],[34, 50],[29, 50],[29, 47.5]]]
        }
      }
    ]
  };

  function alertStyle(feature) {
    const type = feature.properties.type;
    let color = 'red';
    if (type === 'drone') color = 'orange';
    if (type === 'chemical') color = 'green';
    return {
      color: color,
      weight: 2,
      fillColor: color,
      fillOpacity: 0.25
    };
  }

  const alertsLayer = L.geoJSON(alertsGeojson, {
    style: alertStyle,
    onEachFeature: (feature, layer) => {
      layer.bindPopup(
        `<b>${feature.properties.region}</b><br>Alert: ${feature.properties.type}`
      );
    }
  }).addTo(map);

  // 4. Explosions (dummy points)
  const explosionsGeojson = {
    "type": "FeatureCollection",
    "features": [
      {
        "type": "Feature",
        "properties": {
          "type": "explosion",
          "time": "2026-06-28T18:20",
          "desc": "Reported explosion in city"
        },
        "geometry": {
          "type": "Point",
          "coordinates": [30.5, 49.0]
        }
      }
    ]
  };

  const explosionIcon = L.divIcon({
    className: '',
    html: '<span style="color:red;font-size:18px;">✹</span>',
    iconSize: [18, 18],
    iconAnchor: [9, 9]
  });

  const explosionsLayer = L.geoJSON(explosionsGeojson, {
    pointToLayer: (feature, latlng) => L.marker(latlng, { icon: explosionIcon }),
    onEachFeature: (feature, layer) => {
      layer.bindPopup(
        `<b>${feature.properties.type}</b><br>${feature.properties.time}<br>${feature.properties.desc}`
      );
    }
  }).addTo(map);

  // 5. Layer control
  const overlays = {
    "Territorial control": territorialLayer,
    "Air alerts": alertsLayer,
    "Explosions": explosionsLayer
  };
  L.control.layers(null, overlays).addTo(map);

  // 6. Legend
  const legend = L.control({ position: 'bottomright' });
  legend.onAdd = function () {
    const div = L.DomUtil.create('div', 'legend');
    div.innerHTML = `
      <b>Legend</b><br>
      <span style="background:#d73027;"></span> Occupied<br>
      <span style="background:#1a9850;"></span> Liberated<br>
      <span style="background:#fdae61;"></span> Contested<br>
      <span style="background:red;"></span> Missile alert<br>
      <span style="background:orange;"></span> Drone alert<br>
      <span style="background:green;"></span> Chemical alert<br>
      ✹ Explosion report
    `;
    return div;
  };
  legend.addTo(map);
</script>
</body>
